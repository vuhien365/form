Lấy giá trị trong form bằng JavaScript thuần

Chú ý: Tên id hoặc class để lấy value cho đúng
Copy 
                    
                    function showInfo(){ 

                        //lấy giá trị trong thẻ input có id='name' 

                        var uname = document.getElementById("name").value; 

                
                        // Lấy giá trị trong thẻ input có class='password' 

                        var pass = document.getElementsByClassName("password")[0].value; 

                
                        // Lấy giá trị repassword co class='password' 

                        var repass = document.getElementsByClassName("password")[1].value; 

                
                        // Lấy giá trị cho email có id="email" 

                        var uemail = document.getElementById("email").value; 

                
                        // Lấy giá trị cho birthday có id='Birthday'

                        var ubirthday = document.getElementById("birthday").value; // Biến kiểu chuỗi

                
                        //Lấy giá trị tuổi hiện tại

                        var today = new Date(); // biến kiểu Date

                        var birth = new Date(ubirthday); // thay kiểu chuỗi thành kiểu Date 

                        var uage = today.getFullYear() - birth.getFullYear();

                
                        // Lấy giá trị giới tính

                        var gender = document.getElementsByClassName("gender"); //Kiểu array HTMLCollection

                        // Tạo 1 array rỗng

                        var genderArray = new Array();

                        // Dò tìm thuộc tính

                        for (var i in gender){

                            if (gender[i].checked){

                                genderArray.push(gender[i].value);

                            }

                        }

                
                        // Lấy giá trị Sở thích

                        var hob = document.getElementsByClassName("hob"); //chuyển kiểu
                        // Tạo array rỗng

                        var hobArr = [];

                        // Dò tìm thuộc tính

                        for(var i in hob){

                            if (hob[i].checked){

                                hobArr.push(hob[i].value);

                            }

                        }

                
                        // Xuất lên bảng thông báo

                        alert(  " Uname: " + uname + "\n" + 

                                " Pass: " + pass + "\n" + 

                                " Re-Pass: " + repass + "\n" + 

                                " Email: " + uemail + "\n" +

                                " Birthday: " + ubirthday + "\n" +

                                " Số tuổi: " + uage + "\n" +

                                " Giới tính: " + genderArray + "\n" +

                                " Sở thích: " + hobArr

                                );

                        // Xuất thông tin vào khung div
  
      document.getElementById("info").innerHTML = 
  " <h2> Thông tin bạn đã nhập</h2>" + "<br>" +
  
                                                   " Uname: " + uname + "<br>" + 
               
                                      " Pass: " + pass + "<br>" + 
                
                                     " Re-Pass: " + repass + "<br>" + 
               
                                      " Email: " + uemail + "<br>" +
                 
                                    " Birthday: " + ubirthday + "<br>" +
                        
                             " Số tuổi: " + uage + "<br>" +
                        
                             " Giới tính: " + genderArray + "<br>" +
                    
                                 " Sở thích: " + hobArr 

                        // Không cho chuyển trang sau khi submit 

                        return false;

                    }

Code mẫu cho Username, bạn nhớ sửa lại cho đúng id và các thông tin khai báo
Để trong hàm function(){ copy vào đây }
Copy 
                    
                    var name_name = document.getElementById("username"); 

                    var name_value = document.getElementById("username").value;

                    var name_length = name_value.length;

                    var name_pattern = /^[a-z_][a-z0-9_\.]{4,19}$/;

                        if(name_length == ""){

                            alert("Bạn chưa nhập gì cả");

                        }

                        else if(name_length < 5){

                            alert("Bạn nhập dưới 5 ký tự");

                        }

                        else if(name_length > 20){

                            alert("Bạn nhập nhiều hơn 20 ký tự");

                        }

                        else if(!name_value.match(name_pattern)){

                            alert(  "Tên đăng nhập bắt đầu bằng 1 ký tự a-z hoặc _" + "\n" +

                                    "Tên đăng nhập nằm trong tập hợp kí tự a-z, 0-9 và có thể chứa dấu . hoặc _" + "\n" +

                                    "Có Tối thiểu 5 kí tự, và tối đa 20 kí tự"

                                );

                        }

                        else{

                            alert("Thông tin bạn nhập vào hợp lệ! ");

                        }
